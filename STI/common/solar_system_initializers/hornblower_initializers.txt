ufop_outpost_initializer = {
	class = "rl_all_stars"
	flags = { outpost ufop_system ufop }
	usage = static_system_init

	planet = {
		class = star
		orbit_distance = 0
		orbit_angle = 0
		size = { min = 20 max = 30 }
		has_ring = no
	}
	
	change_orbit = 50
	
	planet = {
		count = { min = 3 max = 8 }
		class = rl_unhabitable_planets
		orbit_distance = { min = 20 max = 30 }
		home_planet = no
	}

	late_init_effect = {

		# We force a hyperlane between this system and the main system.
		# If it doesn't have an hyperlane it will attach all the systems between them to the Empire.
		# If the hyperlane feature is removed, this implementation has to be removed.
		#random_system = {
		#	limit = { has_star_flag = ufop_system }
		#	save_event_target_as = ufop_system
		#}
		#if = {
		#	limit = {
		#		NOT = { has_hyperlane_to = event_target:ufop_system }
		#	}
		#	_hyperlane = { from = this to = event_target:ufop_system }
		#}

		# Creating a Starbase
		create_starbase = {
			size = starbase_outpost # We can use any Starbase size. The sizes are located on "game\common\starbase_types"
			owner = event_target:ufop_country # Using the event_target saved on ufop_system_initializer
